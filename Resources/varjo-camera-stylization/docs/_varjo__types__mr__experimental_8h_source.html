38 typedef int64_t varjo_ShaderInputLayout;
39 /*
40  * This layout has these built-in inputs and outputs:
41  * - Input RGBA texture that has the current video see through image.
42  * - Output RGBA texture that should be written from the shader.
43  * - Constant buffer with the following data:
44  * - Texture dimensions.
45  * - Source texture timestamp.
46  * - Currently rendered view index: 0 Left context, 1 Right Context,
47  * 2 Left focus, 3 Right focus.
48  * - Destination rect. Area to render during a particular call to the shader.
49  * Source texture is guarateed to have have up-to-date data on this area.
50  * - Linear clamping texture sampler
51  * - Linear wrapping texture sampler
52  *
53  * HLSL register definitions:
54  *
55  * // Input buffer
56  * Texture2D<float4> inputTex : register(t0);
57  * // Output buffer
58  * RWTexture2D<float4> outputTex : register(u0);
59  * // Varjo generic constants
60  * cbuffer ConstantBuffer : register(b0)
61  * {
62  * int2 textureSize; // Texture dimensions
63  * float sourceTime; // Source texture timestamp
64  * int viewIndex; // View to be rendered: 0=LC, 1=RC, 2=LF, 3=RF
65  * int4 destRect; // Destination rectangle: x, y, w, h
66  * }
67  *
68  * // Varjo generic texture samplers
69  * SamplerState SamplerLinearClamp : register(s0);
70  * SamplerState SamplerLinearWrap : register(s1);
71  */
