46 static void varjo_UpdateNearFarPlanes(double* projectionMatrix, varjo_ClipRange clipRange, double nearZ, double farZ)
47 {
48  if (projectionMatrix == NULL) {
49  return;
50  }
51 
52  if (clipRange == varjo_ClipRangeZeroToOne) {
53  if (farZ == HUGE_VAL) {
54  // Normal Z-range, infinite far clip plane, D3D/Vulkan convention
55  projectionMatrix[10] = -1.;
56  projectionMatrix[14] = -nearZ;
57  } else if (nearZ == HUGE_VAL) {
58  // Inverted Z-range, infinite far clip plane, D3D/Vulkan convention
59  projectionMatrix[10] = 0.;
60  projectionMatrix[14] = farZ;
61  } else {
62  // Note: Following formula handles both cases: when farZ > nearZ and when nearZ > farZ
63  projectionMatrix[10] = farZ / (nearZ - farZ);
64  projectionMatrix[14] = (nearZ * farZ) / (nearZ - farZ);
65  }
66  } else {
67  if (nearZ == HUGE_VAL || farZ == HUGE_VAL) {
68  // Handles both normal and reverted Z-range in OpenGL convention in
69  // the case of far clip plane located at +infinity
70  projectionMatrix[10] = -1.;
71  projectionMatrix[14] = nearZ <= farZ ? nearZ : farZ;
72  } else {
73  // Handles both normal and reverted Z-ranges in OpenGL convention
74  projectionMatrix[10] = (nearZ + farZ) / (nearZ - farZ);
75  projectionMatrix[14] = (2 * nearZ * farZ) / (nearZ - farZ);
76  }
77  }
78 }
